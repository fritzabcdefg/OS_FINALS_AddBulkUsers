$DomainUPN   = "bsit.com"
$Password    = ConvertTo-SecureString "Student12345" -AsPlainText -Force
$Container   = "OU=Students,DC=bsit,DC=com"

Import-Module ActiveDirectory

if (-not (Get-ADOrganizationalUnit -Filter "Name -eq 'Students'")) {
    New-ADOrganizationalUnit -Name "Students" -Path "DC=mydomain,DC=local"
    Write-Host "Created OU: Students"
}

$files = Get-ChildItem "C:\Users\Administrator\Desktop\*.txt"

foreach ($file in $files) {
    $SectionName = $file.BaseName
    Write-Host "Processing file: $($file.Name) → Group: $SectionName"

    if (-not (Get-ADGroup -Filter "Name -eq '$SectionName'")) {
        New-ADGroup -Name $SectionName -GroupCategory Security -GroupScope Global -Path $Container
        Write-Host "Created group: $SectionName"
    }

    $students = Get-Content $file.FullName
    foreach ($line in $students) {
        if (-not $line -or $line.Trim() -eq "") {
            Write-Host "Skipping empty line in $($file.Name)"
            continue
        }

        $parts = $line.Split(',')
        if ($parts.Count -lt 2) {
            Write-Host "Skipping badly formatted line: $line"
            continue
        }

        $lastName       = $parts[0].Trim()
        $firstNameFull  = $parts[1].Trim()
        $firstNameOnly  = $firstNameFull.Split(' ')[0]

        $samAccountName = ($firstNameOnly + "." + $lastName).ToLower()
        $i = 1
        while (Get-ADUser -Filter "SamAccountName -eq '$samAccountName'") {
            $samAccountName = ($firstNameOnly + "." + $lastName + $i).ToLower()
            $i++
        }

        $userPrincipalName = $samAccountName + "@" + $DomainUPN

        try {
            New-ADUser -Name "$firstNameFull $lastName" `
                       -GivenName $firstNameOnly `
                       -Surname $lastName `
                       -DisplayName "$firstNameFull $lastName" `
                       -SamAccountName $samAccountName `
                       -UserPrincipalName $userPrincipalName `
                       -AccountPassword $Password `
                       -Enabled $true `
                       -Path $Container `
                       -Description "Section: $SectionName" `
                       -ChangePasswordAtLogon $true

            Add-ADGroupMember -Identity $SectionName -Members $samAccountName
            Write-Host "Added: $samAccountName to $SectionName"
        }
        catch {
            Write-Host "Error creating or adding user $samAccountName: $_"
        }
    }
}
